SELECT T.PARTITION,A.WORKCENTER,P.MATERIAL,R.NDEVIRHIZI,(R.NDEVIRHIZI)*T.PARTITION*420 AS OPTIMALMIKTAR
FROM IASAUTSTATUS A
LEFT JOIN IASPRDOPR P ON P.CONFIRMATION = A.CONFIRMATION
LEFT JOIN IASPRDRST T ON P.PRDORDER = T.PRDORDER
LEFT JOIN IASROUOPR R ON P.OPERATION = R.OPERATION
AND P.MATERIAL = R.MATERIAL
WHERE A.WORKCENTER = 'P-12';
